```python
        r"–ò–ò-–∞–Ω–∞–ª–∏–∑ –Ω–µ–¥–æ—Å—Ç—É–ø–µ–Ω:.*",
        r"Error code: \d+.*?\}",
        r"Traceback.*",
    ]
    for pattern in error_patterns:
        text = re.sub(pattern, "", text, flags=re.DOTALL)
    return text.strip()
```
